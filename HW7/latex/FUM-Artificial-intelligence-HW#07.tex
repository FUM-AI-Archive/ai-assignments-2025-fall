\section{سوال چهارم}

گزاره‌های زیر را به منطق مرتبه اول تبدیل کنید و سپس نتیجه را با کمک Resolution استنتاج کنید.

\subsection*{الف) مسئله سگ و گربه}

\subsubsection*{گزاره‌ها:}
\begin{enumerate}
	\item همه‌ی سگ‌های شکاری در شب زوزه می‌کشند. 
	\item هر کسی که گربه داشته باشد، موش نخواهد داشت.
	\item افراد کم‌خواب از هیچ موجودی که شب‌ها زوزه می‌کشد نگهداری نمی‌کنند.
	\item جان یا یک گربه دارد یا یک سگ شکاری.
	\item \textbf{(نتیجه)} اگر جان فردی کم‌خواب باشد، آنگاه جان هیچ موشی نخواهد داشت.
\end{enumerate}

\subsection*{ب) مسئله سانتا و کریسمس}

\subsubsection*{گزاره‌ها:}
\begin{enumerate}
	\item هر کودکی هر کسی را که به او هدیه‌ای بدهد دوست دارد.
	\item هر کودکی اگر سانتا بتواند در شب کریسمس سفر کند، از سانتا هدیه‌ای دریافت خواهد کرد.
	\item در شب کریسمس هوا مه‌آلود است.
	\item هر زمان که هوا مه‌آلود باشد، هر کسی می‌تواند سفر کند اگر منبع نوری داشته باشد.
	\item هر گوزنی که بینی قرمز داشته باشد، یک منبع نور است.
	\item \textbf{(نتیجه)} اگر سانتا گوزنی با بینی قرمز داشته باشد، آنگاه هر کودکی سانتا را دوست خواهد داشت.
\end{enumerate}
