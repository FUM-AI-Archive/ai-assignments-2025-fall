\section{سوال اول}

هر یک از گزاره های زیر را اثبات کنید.
\begin{latin} 
\begin{enumerate}
    \item $\alpha$ is valid if and only if $\text{True} \models \alpha$.
    \item For any $\alpha$, $\text{False} \models \alpha$.
    \item $\alpha \models \beta$ if and only if the sentence $(\alpha \Rightarrow \beta)$ is valid.
    \item $\alpha \equiv \beta$ if and only if the sentence $(\alpha \Leftrightarrow \beta)$ is valid.
    \item $\alpha \models \beta$ if and only if the sentence $(\alpha \wedge \neg \beta)$ is unsatisfiable.
\end{enumerate}
\end{latin}

\section{سوال دوم}
الف.
تمامی عبارات زیر را به یک فرم CNF تبدیل کنید.
\begin{latin} 
\begin{enumerate}
    \item $A \Leftrightarrow (C \vee E)$
    \item $E \Rightarrow D$
    \item $B \wedge F \Rightarrow \neg C$
    \item $E \Rightarrow C$
    \item $C \Rightarrow F$
    \item $C \Rightarrow B$
\end{enumerate}
\end{latin}
ب.
با استفاده از روش \textbf{resolution}، جمله
\[
\neg A \wedge \neg B
\]
را از صورت‌های حاصل از بخش الف اثبات کنید.

\section{سوال سوم}

درستی یا نادرستی گزاره های زیر را اثبات کنید.\\
(نکته : به فرم نرمال CNF تبدیل کنید)
\begin{align*}
\quad &(A \rightarrow B) \,\land\, (\neg A \rightarrow C) \,\land\, (B \lor C \rightarrow D) \models D \\[6pt]
\quad &(A \lor B) \,\land\, (\neg A \lor C) \,\land\, (\neg B \lor D) \,\land\, (\neg C \lor G) \,\land\, (\neg D \lor G) \models G \\[6pt]
\quad &A \,\land\, (A \rightarrow B) \,\land\, ((A \rightarrow B) \rightarrow (B \rightarrow C)) \models C \\[6pt]
\quad &\neg A \,\land\, (A \rightarrow B) \models \neg B
\end{align*}

\section{سوال چهارم}

عبارات زیر را به گزاره‌های منطق مرتبهٔ اول تبدیل کنید.
\begin{latin}
\begin{enumerate}

\item 
There is a barber who shaves all and only those who do not shave themselves.\\
\textbf{Predicates:} $Bx$: $x$ is a barber;\; $Sxy$: $x$ shaves $y$.

\item 
Unless there is a creator of everything, everything creates itself.\\
\textbf{Predicate:} $Cxy$: $x$ creates $y$.

\item 
There are smart students and hard-working students.\\
\textbf{Predicates:} $Sx$: $x$ is a student;\; $Mx$: $x$ is smart;\; $Hx$: $x$ is hard-working.

\item 
No less than three rabbits are in John’s garden.\\
\textbf{Predicates:} $Rx$: $x$ is a rabbit;\; $Ixy$: $x$ is in $y$;\; $g$ = John’s garden.


\end{enumerate}
\end{latin}
